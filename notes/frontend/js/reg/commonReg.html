# 常用正则表达式

# 验证手机号（中国）

# 宽松，只要是13,14,15,16,17,18,19开头即可

// 正则
const reg = /^(?:(?:\+|00)86)?1[3-9]\d{9}$/g
// 验证方法
const isMPRelaxed = value => reg.test(value);
